Dados simulados

Code 
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt

rng = np.random.default_rng(42)
n = 50

df = pd.DataFrame({
    "receita": rng.uniform(100, 500, size=n),
    "custo": rng.uniform(50, 300, size=n),
    "canal": rng.choice(["Organico", "Pago", "Indicacao"], size=n),
})
df["lucro"] = df["receita"] - df["custo"]

Figura: distribuição do lucro por canal

Code 
fig, ax = plt.subplots()
for canal, dados in df.groupby("canal"):
    ax.hist(dados["lucro"], alpha=0.6, label=canal, bins=12)
ax.set_xlabel("Lucro")
ax.set_ylabel("Frequência")
ax.legend()
plt.tight_layout()
plt.show()

Figura: receita vs custo

Code 
fig, ax = plt.subplots()
cores = {"Organico": "#2ecc71", "Pago": "#e74c3c", "Indicacao": "#3498db"}
for canal, dados in df.groupby("canal"):
    ax.scatter(dados["custo"], dados["receita"], label=canal, color=cores[canal], alpha=0.7)
ax.set_xlabel("Custo")
ax.set_ylabel("Receita")
ax.legend()
plt.tight_layout()
plt.show()
